[docs]    def save_object(self, obj):
        con = self.get_conn()
        cur = con.cursor()
        accession = obj.accession
        json = obj.json()
        cur.execute("""insert into tbl (accession, json) values (%s, %s)
                     on conflict (accession) do update set json = tbl.json""",
                    (accession, json))
        con.commit()

        con = self.get_conn()
        cur = con.cursor()
        try:
            cur.execute(
                """create table tbl (
                   accession varchar(30) primary key,
                   json JSONB)"""
            )
            con.commit()
        except:
            con.rollback()
